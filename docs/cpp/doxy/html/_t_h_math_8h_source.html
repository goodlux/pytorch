96 static inline double TH_erfinv(double y) {
97 /* Function to calculate inverse error function. Rational approximation
98 is used to generate an initial approximation, which is then improved to
99 full accuracy by two steps of Newton's method. Code is a direct
100 translation of the erfinv m file in matlab version 2.0.
101 Author: Gary L. Pavlis, Indiana University
102 Date: February 1996
103 */
104  double x,z,num,dem; /*working variables */
105  /* coefficients in rational expansion */
106  double a[4]={ 0.886226899, -1.645349621, 0.914624893, -0.140543331};
107  double b[4]={-2.118377725, 1.442710462, -0.329097515, 0.012229801};
108  double c[4]={-1.970840454, -1.624906493, 3.429567803, 1.641345311};
109  double d[2]={ 3.543889200, 1.637067800};
110  if(fabs(y) > 1.0) return (atof("NaN")); /* This needs IEEE constant*/
111  if(fabs(y) == 1.0) return((copysign(1.0,y))*atof("INFINITY"));
112  if(fabs(y) <= CENTRAL_RANGE){
113  z = y*y;
114  num = (((a[3]*z + a[2])*z + a[1])*z + a[0]);
115  dem = ((((b[3]*z + b[2])*z + b[1])*z +b[0])*z + 1.0);
116  x = y*num/dem;
117  }
118  else{
119  z = sqrt(-log((1.0-fabs(y))/2.0));
120  num = ((c[3]*z + c[2])*z + c[1])*z + c[0];
121  dem = (d[1]*z + d[0])*z + 1.0;
122  x = (copysign(1.0,y))*num/dem;
123  }
124  /* Two steps of Newton-Raphson correction */
125  x = x - (erf(x) - y)/( (2.0/sqrt(M_PI))*exp(-x*x));
126  x = x - (erf(x) - y)/( (2.0/sqrt(M_PI))*exp(-x*x));
127 
128  return(x);
129 }
130 #undef CENTRAL_RANGE

251 static inline double TH_trigamma(double x) {
252  double sign = +1;
253  double result = 0;
254  if (x < 0.5) {
255  sign = -1;
256  const double sin_pi_x = sin(M_PI * x);
257  result -= (M_PI * M_PI) / (sin_pi_x * sin_pi_x);
258  x = 1 - x;
259  }
260  for (int i = 0; i < 6; ++i) {
261  result += 1 / (x * x);
262  x += 1;
263  }
264  const double ixx = 1 / (x*x);
265  result += (1 + 1 / (2*x) + ixx * (1./6 - ixx * (1./30 - ixx * (1./42)))) / x;
266  return sign * result;
