31 // Note [Empty versus 0-dim tensors]
32 // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
33 // Unlike Torch, ATen treats zero-dimension tensors as having ONE
34 // element (that is to say, a zero-dimensional tensor is a scalar!)
35 // This is in contrast to Torch, where a zero-dimension tensor has
36 // zero elements.
37 //
38 // Because we are backed by Torch tensors, we need to be able to
39 // represent this state (of numel==0). These tensors are represented
40 // by one-dimensional tensors with size[0] == 0 and stride[0] == 1
41 // (the stride is arbitrary but matches the NumPy equivalent).
42 constexpr std::array<int64_t, 1> kEmptySizes { {0} };
43 constexpr std::array<int64_t, 1> kEmptyStrides { {1} };
