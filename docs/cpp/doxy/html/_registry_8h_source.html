53  void Register(const SrcType& key, Creator creator) {
54  // The if statement below is essentially the same as the following line:
55  // CHECK_EQ(registry_.count(key), 0) << "Key " << key
56  // << " registered twice.";
57  // However, CHECK_EQ depends on google logging, and since registration is
58  // carried out at static initialization time, we do not want to have an
59  // explicit dependency on glog's initialization function.
60  std::lock_guard<std::mutex> lock(register_mutex_);
61  if (registry_.count(key) != 0) {
62  printf("Key already registered.\n");
63  PrintOffendingKey(key);
64  std::exit(1);
65  }
66  registry_[key] = creator;
67  }
