7 // Kernel for fast unfold+copy
8 // (borrowed from Caffe: https://github.com/BVLC/caffe/blob/master/src/caffe/layers/conv_layer.cu)
9 template <typename Dtype>
10 __launch_bounds__(CUDA_NUM_THREADS)
11 __global__ void im2col_kernel(const int64_t n, const Dtype* data_im,
12  const int64_t height, const int64_t width,
13  const int64_t ksize_h, const int64_t ksize_w,
14  const int64_t pad_h, const int64_t pad_w,
15  const int64_t stride_h, const int64_t stride_w,
16  const int64_t dilation_h, const int64_t dilation_w,
17  const int64_t height_col, const int64_t width_col,
18  Dtype* data_col) {
19  CUDA_KERNEL_LOOP(index, n) {
20  int64_t w_out = index % width_col;
21  index /= width_col;
22  int64_t h_out = index % height_col;
23  int64_t channel_in = index / height_col;
24  int64_t channel_out = channel_in * ksize_h * ksize_w;
25  int64_t h_in = h_out * stride_h - pad_h;
26  int64_t w_in = w_out * stride_w - pad_w;
27  data_col += (channel_out * height_col + h_out) * width_col + w_out;
28  data_im += (channel_in * height + h_in) * width + w_in;
29  for (int64_t i = 0; i < ksize_h; ++i) {
30  for (int64_t j = 0; j < ksize_w; ++j) {
31  int64_t h = h_in + i * dilation_h;
32  int64_t w = w_in + j * dilation_w;
33  *data_col = (h >= 0 && w >= 0 && h < height && w < width) ?
34  data_im[i * dilation_h * width + j * dilation_w] : ScalarConvert<int, Dtype>::to(0);
35  data_col += height_col * width_col;
36  }
37  }
38  }
39 }

61 template <typename Dtype, typename Acctype>
62 __launch_bounds__(CUDA_NUM_THREADS)
63 __global__ void col2im_kernel(const int64_t n, const Dtype* data_col,
64  const int64_t height, const int64_t width, const int64_t channels,
65  const int64_t kernel_h, const int64_t kernel_w,
66  const int64_t pad_h, const int64_t pad_w,
67  const int64_t stride_h, const int64_t stride_w,
68  const int64_t dilation_h, const int64_t dilation_w,
69  const int64_t height_col, const int64_t width_col,
70  Dtype* data_im) {
71  CUDA_KERNEL_LOOP(index, n) {
72  Acctype val = Acctype(0);
73  const int64_t w_im = index % width + pad_w;
74  const int64_t h_im = (index / width) % height + pad_h;
75  const int64_t c_im = index / (width * height);
76  int64_t kernel_extent_w = (kernel_w - 1) * dilation_w + 1;
77  int64_t kernel_extent_h = (kernel_h - 1) * dilation_h + 1;
78  // compute the start and end of the output
79  const int64_t w_col_start =
80  (w_im < kernel_extent_w) ? 0 : (w_im - kernel_extent_w) / stride_w + 1;
81  const int64_t w_col_end = min(w_im / stride_w + 1, width_col);
82  const int64_t h_col_start =
83  (h_im < kernel_extent_h) ? 0 : (h_im - kernel_extent_h) / stride_h + 1;
84  const int64_t h_col_end = min(h_im / stride_h + 1, height_col);
85  // TODO: use LCM of stride and dilation to avoid unnecessary loops
86  for (int64_t h_col = h_col_start; h_col < h_col_end; h_col += 1) {
87  for (int64_t w_col = w_col_start; w_col < w_col_end; w_col += 1) {
88  int64_t h_k = (h_im - h_col * stride_h);
89  int64_t w_k = (w_im - w_col * stride_w);
90  if (h_k % dilation_h == 0 && w_k % dilation_w == 0) {
91  h_k /= dilation_h;
92  w_k /= dilation_w;
93  int64_t data_col_index = (((c_im * kernel_h + h_k) * kernel_w + w_k) *
94  height_col + h_col) * width_col + w_col;
95  val += data_col[data_col_index];
96  }
97  }
98  }
99  data_im[index] = ScalarConvert<Acctype, Dtype>::to(val);
100  }
101 }
