10 { \
11  int shape_check_flag = 0; \

13  { \
14  if (TH_TENSOR_DIM_APPLY_i == DIMENSION) \
15  continue; \

17  shape_check_flag = 1; \
18  break; \
19  } \
20  if(TENSOR1->size(TH_TENSOR_DIM_APPLY_i) != TENSOR3->size(TH_TENSOR_DIM_APPLY_i)) { \
21  shape_check_flag = 1; \
22  break; \
23  } \
24  } \
25  if (shape_check_flag == 1) { \
26  AT_ERROR("Expected ", #TENSOR1, " ", TENSOR1->sizes(), ", ", #TENSOR2, " ", TENSOR2->sizes(), " and ", #TENSOR3, " ", TENSOR3->sizes(), " to have the same size apart from dimension ", DIMENSION); \
27  } \
28 }

261 #define TH_TENSOR_DIM_APPLY(TYPE, TENSOR, DIMENSION, CODE) \
262 { \
263  TYPE *TENSOR##_data = NULL; \
264  int64_t TENSOR##_stride = 0, TENSOR##_size = 0; \
265  int64_t *TH_TENSOR_DIM_APPLY_counter = NULL; \
266  int TH_TENSOR_DIM_APPLY_hasFinished = 0; \
267  int TH_TENSOR_DIM_APPLY_i; \
268 \
269  if( (DIMENSION < 0) || (DIMENSION >= TENSOR->_dim()) ) \
270  THError("invalid dimension"); \
271 \
272  TENSOR##_data = THTensor_getStoragePtr(TENSOR)->data<TYPE>()+(TENSOR)->storage_offset(); \
273  TENSOR##_stride = (TENSOR)->stride(DIMENSION); \
274  TENSOR##_size = TENSOR->size(DIMENSION); \
275  /* Counter stores the indices into the Tensor at any time */ \
276  TH_TENSOR_DIM_APPLY_counter = (int64_t*)THAlloc(sizeof(int64_t)*(TENSOR->_dim())); \
277  for(TH_TENSOR_DIM_APPLY_i = 0; TH_TENSOR_DIM_APPLY_i < TENSOR->_dim(); TH_TENSOR_DIM_APPLY_i++) \
278  TH_TENSOR_DIM_APPLY_counter[TH_TENSOR_DIM_APPLY_i] = 0; \
279 \
280  while(!TH_TENSOR_DIM_APPLY_hasFinished) \
281  { \
282  CODE \
283 \
284  if(TENSOR->_dim() == 1) \
285  break; \
286  \
287  for(TH_TENSOR_DIM_APPLY_i = 0; TH_TENSOR_DIM_APPLY_i < TENSOR->_dim(); TH_TENSOR_DIM_APPLY_i++) \
288  { \
289  /* Check if the index is equal to DIMENSION. We don't need to update the */ \
290  /* offset if this is the case, and can consider the next index. However, */ \
291  /* in the case that the DIMENSION is the last index in the Tensor, then */ \
292  /* we have parsed the entire tensor and can exit */ \
293  if(TH_TENSOR_DIM_APPLY_i == DIMENSION) \
294  { \
295  if(TH_TENSOR_DIM_APPLY_i == TENSOR->_dim()-1) \
296  { \
297  TH_TENSOR_DIM_APPLY_hasFinished = 1; \
298  break; \
299  } \
300  continue; \
301  } \
302 \
303  /* Bump the counter at this index, update the pointer */ \
304  TH_TENSOR_DIM_APPLY_counter[TH_TENSOR_DIM_APPLY_i]++; \
305  TENSOR##_data += TENSOR->stride(TH_TENSOR_DIM_APPLY_i); \
306 \
307  if(TH_TENSOR_DIM_APPLY_counter[TH_TENSOR_DIM_APPLY_i] == TENSOR->size(TH_TENSOR_DIM_APPLY_i)) \
308  { \
309  /* Handled TENSOR_size(dim) iterations for DIM_APPLY_i. If this is the last dimension, exit */ \
310  if(TH_TENSOR_DIM_APPLY_i == TENSOR->_dim()-1) \
311  { \
312  TH_TENSOR_DIM_APPLY_hasFinished = 1; \
313  break; \
314  } \
315  else \
316  { \
317  /* Reset the counter, and the pointer to the beginning of the storage for this combination of indices */ \
318  TENSOR##_data -= TH_TENSOR_DIM_APPLY_counter[TH_TENSOR_DIM_APPLY_i]*TENSOR->stride(TH_TENSOR_DIM_APPLY_i); \
319  TH_TENSOR_DIM_APPLY_counter[TH_TENSOR_DIM_APPLY_i] = 0; \
320  } \
321  } \

324  } \
325  } \

327 }
