16 // Tensor is a "generic" object holding a pointer to the underlying TensorImpl object, which
17 // has an embedded reference count. In this way, Tensor is similar to boost::intrusive_ptr.
18 //
19 // For example:
20 //
21 // void func(Tensor a) {
22 // Tensor b = a;
23 // ...
24 // }
25 //
26 // In this example, when we say Tensor b = a, we are creating a new object that points to the
27 // same underlying TensorImpl, and bumps its reference count. When b goes out of scope, the
28 // destructor decrements the reference count by calling release() on the TensorImpl it points to.
29 // The existing constructors, operator overloads, etc. take care to implement the correct semantics.
30 //
31 // Note that Tensor can also be NULL, i.e. it is not associated with any underlying TensorImpl, and
32 // special care must be taken to handle this.
33 struct Tensor : public detail::TensorBase {
34  Tensor() : TensorBase() {}
35  Tensor(TensorImpl * self, bool retain) : TensorBase(self, retain) {}
36  Tensor(const TensorBase & rhs) : TensorBase(rhs) {}
37  Tensor(const Tensor & rhs) = default;
38  Tensor(Tensor && rhs) noexcept = default;

40  // reimplemented from TensorBase so the return type is Tensor rather than TensorBase
41  Tensor & operator=(Tensor && rhs) & {
42  rhs.swap(*this);
43  return *this;
44  }
45  Tensor & operator=(Tensor const & rhs) & {
46  //Tensor ctor retains original rhs.pImpl
47  //then rhs.pImpl is swapped with this->pImpl
48  //finally Tensor dtor releases rhs.pImpl, which was originally this->pImpl
49  Tensor(rhs).swap(*this);
50  return *this;
51  }
